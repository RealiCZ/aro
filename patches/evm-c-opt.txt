    /// Records the compute gas used and returns `false` if the limit has been exceeded.
    ///
    /// This is the per-opcode hot path. `record_gas_used` mutates **only** `self.compute_gas`,
    /// so a full `check_limit()` fan-out over all four dimensions is redundant here: the
    /// `data_size`, `kv_update`, and `state_growth` checks cannot have changed since they last
    /// ran, and every site that increases their usage self-guards with `check_limit()` (which
    /// latches any exceed into `has_exceeded_limit`). Hence, when we reach this point with
    /// `has_exceeded_limit == WithinLimit`, those three trackers are guaranteed within limit, so
    /// the only dimension that can newly exceed is `compute_gas`.
    ///
    /// This is byte-identical to the full `check_limit()`: that path only reaches the
    /// per-dimension fan-out when `has_exceeded_limit == WithinLimit`, and given the invariant
    /// the `data_size`/`kv_update` checks (which have priority) return `WithinLimit` and the
    /// fan-out falls through to the same `compute_gas.check_limit()` and latches the same value.
    ///
    /// INVARIANT (pinned by the `debug_assert!` below and the differential probe): on entry,
    /// `has_exceeded_limit.within_limit()` implies all of `data_size`, `kv_update`, and
    /// `state_growth` are within limit.
    pub(crate) fn record_compute_gas(&mut self, compute_gas_used: u64) -> bool {
        self.compute_gas.record_gas_used(compute_gas_used);

        // Already-latched exceed: identical to `check_limit()`'s short-circuit at entry.
        if self.has_exceeded_limit.exceeded_limit() {
            return false;
        }

        // The non-compute dimensions are unchanged since their last self-guarded check and were
        // within limit then (else `has_exceeded_limit` would be set above), so only `compute_gas`
        // can newly exceed.
        debug_assert!(
            self.data_size.check_limit().within_limit() &&
                self.kv_update.check_limit().within_limit() &&
                self.state_growth.check_limit().within_limit(),
            "record_compute_gas fast path: a non-compute limit was exceeded but not latched into \
             has_exceeded_limit — the per-opcode fan-out reduction invariant is violated"
        );

        let compute_gas_check = self.compute_gas.check_limit();
        if compute_gas_check.exceeded_limit() {
            self.has_exceeded_limit = compute_gas_check;
            return false;
        }
        true
    }
