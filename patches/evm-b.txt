    pub(crate) fn exceeds_current_frame_limit(&self, kind: LimitKind) -> LimitCheck {
        if let Some(entry) = self.frame_stack.last() {
            // Invariant: a single frame's `persistent_usage + discardable_usage` cannot
            // overflow u64. Both fields only grow by resource units (compute gas, data
            // bytes, KV ops, state-growth units) that are bounded far below u64::MAX
            // within one transaction, so the `checked_add` in `used()` can never take its
            // panic branch here. Adding via `wrapping_add` drops that dead overflow check
            // (and its panic landing pad) from this per-opcode hot path; the produced value
            // is identical to `used()` under the invariant. The debug assertion pins it so
            // any future accounting change that could overflow trips in tests/dev.
            debug_assert!(
                entry.persistent_usage.checked_add(entry.discardable_usage).is_some(),
                "frame used() overflow: persistent + discardable exceeds u64"
            );
            let used = entry.persistent_usage.wrapping_add(entry.discardable_usage);
            if used.saturating_sub(entry.refund) > entry.limit {
                return LimitCheck::ExceedsLimit { kind, limit: entry.limit, used, frame_local: true };
            }
        }
        LimitCheck::WithinLimit
    }
